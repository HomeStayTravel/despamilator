def parse text
  text.downcase!

  html_tags.each do |tag|
    if text.match(/<\s*#{tag}\W/) || text.match(/<\n*#{tag}\W/) || text.match(/\W#{tag}\s*\//) || text.match(/\W#{tag}\n*\//)
      self.append_score = 0.3
    end
  end
end
